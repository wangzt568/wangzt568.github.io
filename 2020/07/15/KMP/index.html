      public boolean KMPmatch (String p, String s) {
          int [] sublen = samepresurfix(p);
          int i = 0;
          int j = 0;
          while(j < p.length() && i < s.length()) {
              if(p.charAt(j) == s.charAt(i)){
                  i++;
                  j++;
              } else {
                  if(j != 0) {
                      j = sublen[j - 1];
                  } else {
                      i++;
                  }      
              }
          }
          if(j == p.length()) {
              return true;
          }
          
          return false;
      }
